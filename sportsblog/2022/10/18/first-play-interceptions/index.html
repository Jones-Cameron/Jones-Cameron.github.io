##Find first plays from scrimmage in the first half - how many are there? A: We have two different data sets, so in one set we have 6899 plays and in the other we have 7385 plays

first_plays_game <- pbp %>% filter(firstD_by_poss == 1, lag_kickoff_play == 1, drive_number == 1, play_type != 'Penalty')
first_plays_game2 <- pbp %>% filter(game_play_number == 2, play_type != 'Penalty')

##Count the number of play types for first plays from scrimmage to start the game. A: There are 18 play types for the first play in the game, not including the kickoff.

first_play_type2 <- first_plays_game2 %>% group_by(play_type) %>% summarize(count=n()) %>% arrange(desc(count))
